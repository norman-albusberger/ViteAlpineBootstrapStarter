Alpine.js with Bootstrap - Example Page

Text Binding

HTML Code

<div x-data="{ message: 'Hello, World!' }">
    <span x-text="message"></span>
</div>

Event Handling

HTML Code

><div x-data="{ count: 0 }">
    <button @click="count++" class="btn btn-primary">Click me</button>
    <span x-text="'You have clicked me ' + count + ' times.'"></span>
</div>

Looping

HTML Code

<div x-data="{ items: ['Item 1', 'Item 2', 'Item 3'] }">
    <template x-for="item in items" :key="item">
        <div x-text="item"></div>
    </template>
</div>

Conditional Rendering

This is a secret message!

HTML Code

<div x-data="{ showSecret: false }">
    <button @click="showSecret = !showSecret" class="btn btn-warning mb-3">
        Toggle Secret Message
    </button>
    <div x-show="showSecret" class="alert alert-success">
        This is a secret message!
    </div>
</div>

Form Input Binding

HTML Code

<div x-data="{ formData: { username: '', email: '' } }">
    <form>
        <div class="mb-3">
            <label for="username" class="form-label">Username</label>
            <input type="text" id="username" x-model="formData.username" class="form-control">
        </div>
        <div class="mb-3">
            <label for="email" class="form-label">Email</label>
            <input type="email" id="email" x-model="formData.email" class="form-control">
        </div>
        <div class="alert alert-info" x-text="`Username: ${formData.username}, Email: ${formData.email}`"></div>
    </form>
</div>
